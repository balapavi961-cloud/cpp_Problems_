#include <iostream>
using namespace std;

int main() {
    int n, t;
    cin >> n >> t;

    // Special case
    if (n == 1 && t == 10) {
        cout << -1;
        return 0;
    }

    string result = "";

    if (t == 10) {
        // Must end with 0
        result = "1";
        for (int i = 1; i < n; i++)
            result += "0";
    }
    else {
        // Just print n copies of t
        for (int i = 0; i < n; i++)
            result += char('0' + t);
    }

    cout << result;
    return 0;
}
